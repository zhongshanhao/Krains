# 三分查找

算法流程

  • 首先三分x，得到[l, x1]，(x1, x2)，[x2，r]三个区间，若calc(x1) >= calc(x2)，则说明最小值不在[l1, x1] 这个区间，可以排除，反之排除[x2, r]，这里calc(x1)是固定x1的情况下调整y1使得(x1, y1)到所有点距离的最小值。
  • 计算calc(x)也需要一次三分，跟上述一样，三分y得到3个区间[l, y1],(y1, y2),[y2, r],若f(x, y1) >= f(x, y2),说明最小值不在[l, y1]，可以排除，反之排除[y2, r]，f(x, y1)表示点(x, y1)到各点距离之和。

题目要求精度1e-5之内，因此要三分结束条件为1e-6或者更小。

class Solution {
    int[][] points;

    // 计算点(x, y)到每个点的欧几里得距离
    public double getDistSum(double x, double y){
        double ans = 0;

        for(int[] p : points){
            ans += Math.sqrt((x - p[0]) * (x - p[0]) + (y - p[1]) * (y - p[1]));
        }

        return ans;
    }

    // 固定x，三分y，求固定x时最优的y使得(x, y)到其他各点的距离最小
    public double calc(double x){
        double l = 0, r = 100;

        while(r - l > 1e-6){
            double y1 = l + (r - l) / 3;
            double y2 = l + (r - l) / 3 * 2;
            if(getDistSum(x, y1) >= getDistSum(x, y2)){
                l = y1;
            }else{
                r = y2;
            }
        }

        return getDistSum(x, l);
    }

    public double getMinDistSum(int[][] positions) {
        points = positions;
        double l = 0, r = 100;

        while(r - l > 1e-6){
            double x1 = l + (r - l) / 3;
            double x2 = l + (r - l) / 3 * 2;
            if(calc(x1) >= calc(x2)){
                l = x1;
            }else{
                r = x2;
            }
        }

        return calc(l);
    }
}
